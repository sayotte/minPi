# minPi firmware config

# Boot in 64-bit mode
arm_64bit=1

# Use kernel8.img (aarch64)
kernel=kernel8.img

# Load initramfs into RAM
initramfs ramfs.gz followkernel

# Serial console
enable_uart=1

# Let firmware auto-detect and load the correct DTB from boot partition

# GPU memory — increase if using HDMI
gpu_mem=64

# No splash screen
disable_splash=1

# Disable unnecessary hardware at firmware level
dtparam=audio=off
